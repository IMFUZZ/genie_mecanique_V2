

CREATE TABLE outils(
	id TEXT PRIMARY KEY,
	name TEXT NOT NULL,
	description TEXT,
	quantity INT
		NOT NULL,
	available TINYINT 
		DEFAULT 1,
	broken TINYINT
		DEFAULT 0,
	must_command TINYINT
		DEFAULT 0
);
