

CREATE TABLE outils(
	id INTEGER PRIMARY KEY 
		AUTOINCREMENT,
	name TEXT NOT NULL,
	description TEXT,
	available TINYINT 
		DEFAULT 1,
	broken TINYINT
		DEFAULT 0
);

CREATE TABLE bruts(
	id INTEGER PRIMARY KEY 
		AUTOINCREMENT,
	name TEXT NOT NULL,
	description TEXT,
	quantity INT
		NOT NULL,
	available TINYINT 
		DEFAULT 1,
	must_command TINYINT
		DEFAULT 0
);
